@import andxor.syntax._

@(tpes: List[String])
  def build[@{tpes.prod}]: AndXor@{tpes.length}[@{tpes.prod}] = AndXor@{tpes.length}[@{tpes.prod}]
  def buildF[@{tpes.prod}]: AndXorF@{tpes.length}[@{tpes.prod}] = AndXorF@{tpes.length}[@{tpes.prod}]
  def buildK[F[_], @{tpes.prod}]: AndXorK@{tpes.length}[F, @{tpes.prod}] = AndXorK@{tpes.length}[F, @{tpes.prod}]
