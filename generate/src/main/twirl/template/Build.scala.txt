@import andxor.syntax._

@(tpes: List[String])
  def build[@{tpes.tpeParams}]: AndXor@{tpes.length}[@{tpes.tpeParams}] = AndXor@{tpes.length}[@{tpes.tpeParams}]
  def buildF[@{tpes.tpeParams}]: AndXorF@{tpes.length}[@{tpes.tpeParams}] = AndXorF@{tpes.length}[@{tpes.tpeParams}]
  def buildK[F[_], @{tpes.tpeParams}]: AndXorK@{tpes.length}[F, @{tpes.tpeParams}] = AndXorK@{tpes.length}[F, @{tpes.tpeParams}]
